Objects

In JavaScript, objects use named indexes for it's members.


Object Constructor
function person(first, last, age, eye) {
    this.firstName = first;
    this.lastName = last;
    this.age = age;
    this.eyeColor = eye;
}
var myFather = new person("John", "Doe", 50, "blue");
var myMother = new person("Sally", "Rally", 48, "green");


Object Literal
var person = {firstName:"John", lastName:"Doe", age:50, eyeColor:"blue"};

var person = {
    firstName:"John",
    lastName:"Doe",
    age:50,
    eyeColor:"blue"
};

person.lastName;
person["lastName"];


Objects using the keyword new (not recommended)
var person = new Object();
person.firstName = "John";
person.lastName = "Doe";
person.age = 50;
person.eyeColor = "blue";


Objects Methods
var person = {
    firstName: "John",
    lastName : "Doe",
    id       : 5566,
    fullName : function(c) {
       return this.firstName + " " + this.lastName;
    }
};

person.fullName()


Prototype


function PrintStuff (myDocuments) {
​	this.documents = myDocuments;
}
​
​// We add the print () method to PrintStuff prototype property so other instances of objects can inherit it:​
PrintStuff.prototype.print = function () {
	console.log(this.documents);
}
​
​// Create a new object with the PrintStuff () constructor
​var newObj = new PrintStuff ("I am a new Object.");
​
​// newObj inherits print() from PrintStuff function. newObj can call print directly.​
newObj.print ();
